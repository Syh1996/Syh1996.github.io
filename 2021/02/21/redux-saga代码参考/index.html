             const tags = context.querySelectorAll('*');
             const hTagReg = /^H[1-6]$/;
             let template = '';
             tags.forEach(item=>{
                     if(hTagReg.test(item.tagName.toUpperCase()) && item.innerText){
                      const tagH = item.tagName.toLowerCase();
                      const forHash = item.getAttribute("id");
                     template+=`
                          <${tagH}>
                          ${item.innerText}
                           
                      `;
                  }
              });
             articleTitleList.innerHTML=template; 
          }
